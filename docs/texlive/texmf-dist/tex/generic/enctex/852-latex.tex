% 852-latex is able to input DOS files in CP 852 encoding and has the
% internal encoding in ISO8859-2. Output to .log, .aux, .toc is in
% CP 852. This format exactly emulates the cslatex on emTeX system
% where CP 852 tcp table was choosen during initex.bat.

\else %% The encTeX is detected, OK!
   \begingroup
   \catcode`\{=1  \catcode`\}=2 \catcode`\#=6
   \message{cslatex with ISO8859-2 internal and CP 852 input encoding}
   \def\setcode #1 #2 {\xordcode#1=#2 \xchrcode#2=#1 \xprncode#2=1 }
   \xordcode"1A="20  % DOS's EOF (^Z) is mapped to space
   % The DOS's \r (^M) is ignored implicitly, see code texextra.c, line 866
   %
   %          852  il2
   \setcode   "B5  "C1   % \'A
   \setcode   "A0  "E1   % \'a

   \setcode   "90  "C9   % \'E
   \setcode   "82  "E9   % \'e
   \setcode   "B7  "CC   % \v E
   \setcode   "D8  "EC   % \v e
